sqrt1pm1

Compute sqrt( 1 + x ) - 1.

Usage

var sqrt1pm1 = require( '@stdlib/math/base/special/sqrt1pm1' );

sqrt1pm1( x )

Computes sqrt( 1 + x ) - 1 more accurately for small x. 

var v = sqrt1pm1( 3.0 );
// returns 1.0

v = sqrt1pm1( 0.5 );
// returns ~0.225

v = sqrt1pm1( 0.02 );
// returns ~0.01

v = sqrt1pm1( -0.5 );
// returns ~-0.293

v = sqrt1pm1( -1.1 );
// returns NaN

v = sqrt1pm1( NaN );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var round = require( '@stdlib/math/base/special/round' );
var sqrt1pm1 = require( '@stdlib/math/base/special/sqrt1pm1' );

var x;
var i;

for ( i = 0; i < 100; i++ ) {
    x = round( randu() * 100.0 );
    console.log( 'sqrt(1+%d) - 1 = %d', x, sqrt1pm1( x ) );
}
