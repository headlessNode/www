Architecture

Operating system CPU architecture.

Usage

var ARCH = require( '@stdlib/os/arch' );

ARCH

Operating system CPU architecture.

console.log( ARCH );
// => <string>

Examples

var ARCH = require( '@stdlib/os/arch' );

if ( ARCH === 'arm' || ARCH === 'arm64' ) {
    console.log( 'Running on ARM...' );
} else {
    console.log( 'Running on something else...' );
}
