isIteratorLike

Test if a value is iterator-like.

Usage

var isIteratorLike = require( '@stdlib/assert/is-iterator-like' );

isIteratorLike( value )

Tests if a value is iterator-like.

var obj = {
    'next': function noop() {}
};
var bool = isIteratorLike( obj );
// returns true

bool = isIteratorLike( {} );
// returns false

Notes

  • An iterator protocol-compliant object is an object having a next method following the iterator protocol.
  • As full iterator compliance is impossible to achieve without evaluating an iterator, this function checks only for interface compliance.

Examples

var noop = require( '@stdlib/utils/noop' );
var isIteratorLike = require( '@stdlib/assert/is-iterator-like' );

var obj = {
    'next': noop
};
var bool = isIteratorLike( obj );
// returns true

bool = isIteratorLike( {} );
// returns false

bool = isIteratorLike( [] );
// returns false

bool = isIteratorLike( null );
// returns false
