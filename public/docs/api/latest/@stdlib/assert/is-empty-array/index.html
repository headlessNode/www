isEmptyArray

Test if a value is an empty array.

Usage

var isEmptyArray = require( '@stdlib/assert/is-empty-array' );

isEmptyArray( value )

Tests if a value is an empty array.

var bool = isEmptyArray( [] );
// returns true

Examples

var isEmptyArray = require( '@stdlib/assert/is-empty-array' );

var bool = isEmptyArray( [] );
// returns true

bool = isEmptyArray( new Array() );
// returns true

bool = isEmptyArray( [ 1, 2, 3, 4 ] );
// returns false

bool = isEmptyArray( {} );
// returns false
