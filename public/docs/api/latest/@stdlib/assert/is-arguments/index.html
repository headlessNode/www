isArguments

Test if a value is an arguments object.

Usage

var isArguments = require( '@stdlib/assert/is-arguments' );

isArguments( value )

Tests if a value is an arguments object.

function foo() {
    return arguments;
}

var bool = isArguments( foo() );
// returns true

bool = isArguments( [] );
// returns false

Examples

var isArguments = require( '@stdlib/assert/is-arguments' );

function foo() {
    return arguments;
}

var bool = isArguments( foo() );
// returns true

bool = isArguments( [] );
// returns false

bool = isArguments( {} );
// returns false

bool = isArguments( null );
// returns false

bool = isArguments( 'Arguments' );
// returns false

function Arguments() {
    return this;
}
bool = isArguments( new Arguments() );
// returns false
