REPL

REPL environment.

Usage

var repl = require( '@stdlib/repl/server' );

repl( [options], clbk )

Starts a REPL environment.

repl( onStart );

function onStart( error, server ) {
    if ( error ) {
        throw error;
    }
}

Examples

var repl = require( '@stdlib/repl/server' );

// Start the REPL:
repl( onStart );

function onStart( error, server ) {
    if ( error ) {
        throw error;
    }
    // After the REPL has started, exit the process...
    setTimeout( onTimeout, 2500 );

    function onTimeout() {
        server.close();
    }
}
