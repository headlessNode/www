Italian Stop Words

A list of Italian stop words.

Usage

var stopwords = require( '@stdlib/datasets/savoy-stopwords-it' );

stopwords()

Returns a list of 399 Italian stop words.

var list = stopwords();
/* returns
    [
        'a',
        'abbastanza',
        'accidenti',
        'ad',
        'adesso',
        'affinche',
        ...
    ]
*/

Examples

var floor = require( '@stdlib/math/base/special/floor' );
var randu = require( '@stdlib/random/base/randu' );
var stopwords = require( '@stdlib/datasets/savoy-stopwords-it' );

var list = stopwords();
var len = list.length;
var idx;
var i;

// Select random words from the list...
for ( i = 0; i < 100; i++ ) {
    idx = floor( randu()*len );
    console.log( list[ idx ] );
}
