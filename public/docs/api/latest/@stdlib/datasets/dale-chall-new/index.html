Dale-Chall

A list of familiar American-English words.

Usage

var words = require( '@stdlib/datasets/dale-chall-new' );

words()

Returns a list of familiar American-English words.

var data = words();
/* returns
    [
        'a',
        'able',
        'aboard',
        'about',
        'above',
        ...
    ]
*/

Examples

var floor = require( '@stdlib/math/base/special/floor' );
var randu = require( '@stdlib/random/base/randu' );
var words = require( '@stdlib/datasets/dale-chall-new' );

var data = words();
var len = data.length;
var idx;
var i;

// Select random words from the list...
for ( i = 0; i < 100; i++ ) {
    idx = floor( randu()*len );
    console.log( data[ idx ] );
}
