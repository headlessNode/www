Female First Names

A list of common female first names in English speaking countries.

Usage

var names = require( '@stdlib/datasets/female-first-names-en' );

names()

Returns a list of common female first names (also known as given or personal names) in English speaking countries.

var data = names();
/* returns
    [
        'Aaren',
        'Aarika',
        'Abagael',
        'Abagail',
        'Abbe',
        'Abbey',
        'Abbi',
        'Abbie',
        'Abby',
        ...
    ]
*/

Examples

var floor = require( '@stdlib/math/base/special/floor' );
var randu = require( '@stdlib/random/base/randu' );
var names = require( '@stdlib/datasets/female-first-names-en' );

var data = names();
var len = data.length;
var idx;
var i;

// Select random names from the list...
for ( i = 0; i < 100; i++ ) {
    idx = floor( randu()*len );
    console.log( data[ idx ] );
}
