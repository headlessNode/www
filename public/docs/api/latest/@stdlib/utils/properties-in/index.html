propertiesIn

Return an array of an object's own and inherited property names and symbols.

Usage

var propertiesIn = require( '@stdlib/utils/properties-in' );

propertiesIn( obj )

Returns an array of an object's own and inherited property names and symbols.

var props = propertiesIn( [] );
// returns [...]

Examples

var hasSymbolSupport = require( '@stdlib/assert/has-symbol-support' );
var Symbol = require( '@stdlib/symbol/ctor' );
var propertiesIn = require( '@stdlib/utils/properties-in' );

var hasSymbols;
var props;
var obj;

hasSymbols = hasSymbolSupport();

function Foo() {
    this.a = 'b';
    if ( hasSymbols ) {
        this[ Symbol( 'a' ) ] = 'b';
    }
    return this;
}

Foo.prototype.foo = 'bar';
if ( hasSymbols ) {
    Foo.prototype[ Symbol( 'foo' ) ] = 'bar';
}

obj = new Foo();
props = propertiesIn( obj );

console.log( props );
// e.g., => [ 'a', 'foo', ... ]
