RegExp String

Regular expression to parse a regular expression string.

Usage

var reRegExp = require( '@stdlib/regexp/regexp' );

reRegExp()

Returns a regular expression to parse a regular expression string.

var RE_REGEXP = reRegExp();
var bool = RE_REGEXP.test( '/^beep$/' );
// returns true

bool = RE_REGEXP.test( '' );
// returns false

var parts = RE_REGEXP.exec( '/^.*$/ig' );
// returns [ '/^.*$/ig', '^.*$', 'ig', 'index': 0, 'input': '/^.*$/ig' ]

Regular expression strings should be escaped.

var RE_REGEXP = reRegExp();

var bool = RE_REGEXP.test( '/^\/([^\/]+)\/(.*)$/' );
// returns false

bool = RE_REGEXP.test( '/^\\/([^\\/]+)\\/(.*)$/' );
// returns true

reRegExp.REGEXP

Regular expression to parse a regular expression string.

var bool = reRegExp.REGEXP.test( '/^beep$/' );
// returns true

Examples

var reRegExp = require( '@stdlib/regexp/regexp' );

var RE_REGEXP = reRegExp();

var bool = RE_REGEXP.test( '/beep/' );
// returns true

bool = RE_REGEXP.test( '/^.*$/ig' );
// returns true

bool = RE_REGEXP.test( '/^\\/([^\\/]+)\\/(.*)$/' );
// returns true

bool = RE_REGEXP.test( '/^\/([^\/]+)\/(.*)$/' );
// returns false

bool = RE_REGEXP.test( '/boop' );
// returns false

bool = RE_REGEXP.test( '' );
// returns false
