floor2

Round a numeric value to the nearest power of two toward negative infinity.

Usage

var floor2 = require( '@stdlib/math/base/special/floor2' );

floor2( x )

Rounds a numeric value to the nearest power of two toward negative infinity.

var v = floor2( -4.2 );
// returns -8.0

v = floor2( -4.5 );
// returns -8.0

v = floor2( -4.6 );
// returns -8.0

v = floor2( 9.99999 );
// returns 8.0

v = floor2( 9.5 );
// returns 8.0

v = floor2( 13.0 );
// returns 8.0

v = floor2( -13.0 );
// returns -16.0

v = floor2( 0.0 );
// returns 0.0

v = floor2( -0.0 );
// returns -0.0

v = floor2( Infinity );
// returns Infinity

v = floor2( -Infinity );
// returns -Infinity

v = floor2( NaN );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var floor2 = require( '@stdlib/math/base/special/floor2' );

var x;
var v;
var i;

for ( i = 0; i < 100; i++ ) {
    x = (randu()*100.0) - 50.0;
    v = floor2( x );
    console.log( 'Value: %d. Rounded: %d.', x, v );
}
