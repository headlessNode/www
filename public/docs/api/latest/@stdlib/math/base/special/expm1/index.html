expm1

Compute exp(x) - 1.

Usage

var expm1 = require( '@stdlib/math/base/special/expm1' );

expm1( x )

Computes exp(x) - 1, where exp(x) is the natural exponential function.

var v = expm1( 0.2 );
// returns ~0.221

v = expm1( -9.0 );
// returns ~-1.0

v = expm1( 0.0 );
// returns 0.0

v = expm1( NaN );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var expm1 = require( '@stdlib/math/base/special/expm1' );

var x;
var i;

for ( i = 0; i < 100; i++ ) {
    x = (randu()*10.0) - 5.0;
    console.log( 'e^%d - 1 = %d', x, expm1( x ) );
}
