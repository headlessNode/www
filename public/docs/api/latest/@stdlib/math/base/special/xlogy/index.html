xlogy

Compute x * ln(y) so that the result is 0 if x = 0.

Usage

var xlogy = require( '@stdlib/math/base/special/xlogy' );

xlogy( x, y )

Computes x * ln(y) so that the result is 0 if x = 0.

var out = xlogy( 3.0, 2.0 );
// returns ~2.079

out = xlogy( 1.5, 5.9 );
// returns ~2.662

out = xlogy( 0.9, 1.0 );
// returns 0.0

out = xlogy( 0.0, -2.0 );
// returns 0.0

out = xlogy( 1.5, NaN );
// returns NaN

out = xlogy( 0.0, NaN );
// returns NaN

out = xlogy( NaN, 2.3 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var xlogy = require( '@stdlib/math/base/special/xlogy' );

var x;
var y;
var i;

for ( i = 0; i < 100; i++ ) {
    x = randu();
    if ( x < 0.5 ) {
        x = 0.0;
    }
    y = ( randu() * 20.0 ) - 5.0;
    console.log( 'xlogy(%d, %d) = %d', x, y, xlogy( x, y ) );
}
