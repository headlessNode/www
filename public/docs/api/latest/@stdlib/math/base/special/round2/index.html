round2

Round a numeric value to the nearest power of two on a linear scale.

Usage

var round2 = require( '@stdlib/math/base/special/round2' );

round2( x )

Rounds a numeric value to the nearest power of two on a linear scale.

var v = round2( -4.2 );
// returns -4.0

v = round2( -4.5 );
// returns -4.0

v = round2( -4.6 );
// returns -4.0

v = round2( 9.99999 );
// returns 8.0

v = round2( 9.5 );
// returns 8.0

v = round2( 13.0 );
// returns 16.0

v = round2( -13.0 );
// returns -16.0

v = round2( 0.0 );
// returns 0.0

v = round2( -0.0 );
// returns -0.0

v = round2( Infinity );
// returns Infinity

v = round2( -Infinity );
// returns -Infinity

v = round2( NaN );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var round2 = require( '@stdlib/math/base/special/round2' );

var x;
var v;
var i;

for ( i = 0; i < 100; i++ ) {
    x = (randu()*100.0) - 50.0;
    v = round2( x );
    console.log( 'Value: %d. Rounded: %d.', x, v );
}
