ceil10

Round a numeric value to the nearest power of 10 toward positive infinity.

Usage

var ceil10 = require( '@stdlib/math/base/special/ceil10' );

ceil10( x )

Rounds a numeric value to the nearest power of 10 toward positive infinity.

var v = ceil10( -4.2 );
// returns -1.0

v = ceil10( -4.5 );
// returns -1.0

v = ceil10( -4.6 );
// returns -1.0

v = ceil10( 9.99999 );
// returns 10.0

v = ceil10( 9.5 );
// returns 10.0

v = ceil10( 13.0 );
// returns 100.0

v = ceil10( -13.0 );
// returns -10.0

v = ceil10( 0.0 );
// returns 0.0

v = ceil10( -0.0 );
// returns -0.0

v = ceil10( Infinity );
// returns Infinity

v = ceil10( -Infinity );
// returns -Infinity

v = ceil10( NaN );
// returns NaN

Notes

  • The function may not return accurate results for subnormals due to a general loss in precision.

    var v = ceil10( -1.0e-323 ); // should return -1.0e-323
// returns -0.0

Examples

var randu = require( '@stdlib/random/base/randu' );
var ceil10 = require( '@stdlib/math/base/special/ceil10' );

var x;
var v;
var i;

for ( i = 0; i < 100; i++ ) {
    x = (randu()*100.0) - 50.0;
    v = ceil10( x );
    console.log( 'Value: %d. Rounded: %d.', x, v );
}
