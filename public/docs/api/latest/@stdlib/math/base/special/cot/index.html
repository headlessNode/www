Cotangent

Evaluate the cotangent of a number.

Usage

var cot = require( '@stdlib/math/base/special/cot' );

cot( x )

Evaluates the cotangent of x (in radians).

var v = cot( 0.0 );
// returns Infinity

v = cot( 3.141592653589793/2.0 );
// returns ~0.0

v = cot( -3.141592653589793/4.0 );
// returns ~-1.0

v = cot( 3.141592653589793/4.0 );
// returns ~1.0

v = cot( NaN );
// returns NaN

Examples

var linspace = require( '@stdlib/array/base/linspace' );
var PI = require( '@stdlib/constants/float64/pi' );
var cot = require( '@stdlib/math/base/special/cot' );

var x = linspace( -PI/2.0, PI/2.0, 100 );

var i;
for ( i = 0; i < x.length; i++ ) {
    console.log( cot( x[ i ] ) );
}
