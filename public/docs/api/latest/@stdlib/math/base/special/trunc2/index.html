trunc2

Round a numeric value to the nearest power of two toward zero.

Usage

var trunc2 = require( '@stdlib/math/base/special/trunc2' );

trunc2( x )

Rounds a numeric value to the nearest power of two toward zero.

var v = trunc2( -4.2 );
// returns -4.0

v = trunc2( -4.5 );
// returns -4.0

v = trunc2( -4.6 );
// returns -4.0

v = trunc2( 9.99999 );
// returns 8.0

v = trunc2( 9.5 );
// returns 8.0

v = trunc2( 13.0 );
// returns 8.0

v = trunc2( -13.0 );
// returns -8.0

v = trunc2( 0.0 );
// returns 0.0

v = trunc2( -0.0 );
// returns -0.0

v = trunc2( Infinity );
// returns Infinity

v = trunc2( -Infinity );
// returns -Infinity

v = trunc2( NaN );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var trunc2 = require( '@stdlib/math/base/special/trunc2' );

var x;
var v;
var i;

for ( i = 0; i < 100; i++ ) {
    x = (randu()*100.0) - 50.0;
    v = trunc2( x );
    console.log( 'Value: %d. Rounded: %d.', x, v );
}
