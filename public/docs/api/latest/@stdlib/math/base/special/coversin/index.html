Coversine

Compute the coversed sine.

The coversed sine is defined as

Usage

var coversin = require( '@stdlib/math/base/special/coversin' );

coversin( x )

Computes the coversed sine (in radians).

var v = coversin( 0.0 );
// returns 1.0

v = coversin( 3.141592653589793/2.0 );
// returns 0.0

v = coversin( -3.141592653589793/6.0 );
// returns 1.5

Examples

var linspace = require( '@stdlib/array/linspace' );
var TWO_PI = require( '@stdlib/constants/float64/two-pi' );
var coversin = require( '@stdlib/math/base/special/coversin' );

var x = linspace( 0.0, TWO_PI, 100 );
var i;

for ( i = 0; i < x.length; i++ ) {
    console.log( coversin( x[ i ] ) );
}
