Mean

Normal distribution expected value.

The expected value for a normal random variable with mean μ and standard deviation σ > 0 is

Usage

var mean = require( '@stdlib/stats/base/dists/normal/mean' );

mean( mu, sigma )

Returns the expected value for a normal distribution with parameters mu (mean) and sigma (standard deviation).

var y = mean( 2.0, 1.0 );
// returns 2.0

y = mean( 0.0, 1.0 );
// returns 0.0

y = mean( -1.0, 4.0 );
// returns -1.0

If provided NaN as any argument, the function returns NaN.

var y = mean( NaN, 1.0 );
// returns NaN

y = mean( 0.0, NaN );
// returns NaN

If provided sigma <= 0, the function returns NaN.

var y = mean( 0.0, 0.0 );
// returns NaN

y = mean( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var mean = require( '@stdlib/stats/base/dists/normal/mean' );

var sigma;
var mu;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    sigma = randu() * 20.0;
    y = mean( mu, sigma );
    console.log( 'µ: %d, σ: %d, E(X;µ,σ): %d', mu.toFixed( 4 ), sigma.toFixed( 4 ), y.toFixed( 4 ) );
}
