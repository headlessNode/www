Median

Laplace distribution median.

The median for a Laplace random variable with location parameter mu and scale parameter b > 0 is

Usage

var median = require( '@stdlib/stats/base/dists/laplace/median' );

median( mu, b )

Returns the median for a Laplace distribution with location parameter mu and scale parameter b.

var y = median( 2.0, 1.0 );
// returns 2.0

y = median( 0.0, 1.0 );
// returns 0.0

y = median( -1.0, 4.0 );
// returns -1.0

If provided NaN as any argument, the function returns NaN.

var y = median( NaN, 1.0 );
// returns NaN

y = median( 0.0, NaN );
// returns NaN

If provided b <= 0, the function returns NaN.

var y = median( 0.0, 0.0 );
// returns NaN

y = median( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var median = require( '@stdlib/stats/base/dists/laplace/median' );

var mu;
var b;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    b = randu() * 20.0;
    y = median( mu, b );
    console.log( 'µ: %d, b: %d, Median(X;µ,b): %d', mu.toFixed( 4 ), b.toFixed( 4 ), y.toFixed( 4 ) );
}
