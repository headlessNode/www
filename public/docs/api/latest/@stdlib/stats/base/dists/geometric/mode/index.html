Mode

Geometric distribution mode.

The mode for a geometric random variable with success probability p is

Usage

var mode = require( '@stdlib/stats/base/dists/geometric/mode' );

mode( p )

Returns the mode of a geometric distribution with success probability p.

var v = mode( 0.1 );
// returns 0

v = mode( 0.5 );
// returns 0

If provided a success probability p outside of [0,1], the function returns NaN.

var v = mode( NaN );
// returns NaN

v = mode( 1.5 );
// returns NaN

v = mode( -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var round = require( '@stdlib/math/base/special/round' );
var mode = require( '@stdlib/stats/base/dists/geometric/mode' );

var v;
var i;
var p;

for ( i = 0; i < 10; i++ ) {
    p = randu();
    v = mode( p );
    console.log( 'p: %d, mode(X;p): %d', p.toFixed( 4 ), v.toFixed( 4 ) );
}
