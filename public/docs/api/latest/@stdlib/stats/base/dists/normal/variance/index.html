Variance

Normal distribution variance.

The variance for a normal random variable with mean μ and standard deviation σ > 0 is

Usage

var variance = require( '@stdlib/stats/base/dists/normal/variance' );

variance( mu, sigma )

Returns the variance for a normal distribution with parameters mu (mean) and sigma (standard deviation).

var y = variance( 2.0, 1.0 );
// returns 1.0

y = variance( -1.0, 4.0 );
// returns 16.0

If provided NaN as any argument, the function returns NaN.

var y = variance( NaN, 1.0 );
// returns NaN

y = variance( 0.0, NaN );
// returns NaN

If provided sigma <= 0, the function returns NaN.

var y = variance( 0.0, 0.0 );
// returns NaN

y = variance( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var variance = require( '@stdlib/stats/base/dists/normal/variance' );

var sigma;
var mu;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    sigma = randu() * 20.0;
    y = variance( mu, sigma );
    console.log( 'µ: %d, σ: %d, Var(X;µ,σ): %d', mu.toFixed( 4 ), sigma.toFixed( 4 ), y.toFixed( 4 ) );
}
