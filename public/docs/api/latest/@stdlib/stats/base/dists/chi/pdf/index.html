Usage

var pdf = require( '@stdlib/stats/base/dists/chi/pdf' );

pdf( x, k )

Evaluates the probability density function (PDF) for a chi distribution with degrees of freedom k.

var y = pdf( 0.1, 1.0 );
// returns ~0.794

y = pdf( 0.5, 2.0 );
// returns ~0.441

y = pdf( -1.0, 4.0 );
// returns 0.0

If provided NaN as any argument, the function returns NaN.

var y = pdf( NaN, 1.0 );
// returns NaN

y = pdf( 0.0, NaN );
// returns NaN

If provided k < 0, the function returns NaN.

var y = pdf( 2.0, -2.0 );
// returns NaN

If provided k = 0, the function evaluates the PDF of a degenerate distribution centered at 0.

var y = pdf( 2.0, 0.0 );
// returns 0.0

y = pdf( 0.0, 0.0 );
// returns Infinity

pdf.factory( k )

Returns a function for evaluating the PDF for a chi distribution with degrees of freedom k.

var myPDF = pdf.factory( 6.0 );

var y = myPDF( 3.0 );
// returns ~0.337

y = myPDF( 1.0 );
// returns ~0.076

Examples

var randu = require( '@stdlib/random/base/randu' );
var pdf = require( '@stdlib/stats/base/dists/chi/pdf' );

var k;
var x;
var y;
var i;

for ( i = 0; i < 20; i++ ) {
    x = randu() * 10.0;
    k = randu() * 10.0;
    y = pdf( x, k );
    console.log( 'x: %d, k: %d, f(x;k): %d', x.toFixed( 4 ), k.toFixed( 4 ), y.toFixed( 4 ) );
}
