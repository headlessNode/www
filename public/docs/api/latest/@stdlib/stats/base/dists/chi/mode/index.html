Mode

Chi distribution mode.

The mode for a chi random variable is

for degrees of freedom k >= 1. For other values of k, the mode is not defined.

Usage

var mode = require( '@stdlib/stats/base/dists/chi/mode' );

mode( k )

Returns the mode of a chi distribution with degrees of freedom k.

var v = mode( 9.0 );
// returns ~2.828

v = mode( 1.5 );
// returns ~0.707

If provided k < 1, the function returns NaN.

var v = mode( -1.0 );
// returns NaN

v = mode( 0.9 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var round = require( '@stdlib/math/base/special/round' );
var mode = require( '@stdlib/stats/base/dists/chi/mode' );

var k;
var v;
var i;

for ( i = 0; i < 10; i++ ) {
    k = randu() * 20.0;
    v = mode( k );
    console.log( 'k: %d, mode(X,k): %d', k.toFixed( 4 ), v.toFixed( 4 ) );
}
