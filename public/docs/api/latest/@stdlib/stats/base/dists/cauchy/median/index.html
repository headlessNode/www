Median

Cauchy distribution median.

The median for a Cauchy random variable with location parameter x0 and scale parameter Ɣ > 0 is

Usage

var median = require( '@stdlib/stats/base/dists/cauchy/median' );

median( x0, gamma )

Returns the median of a Cauchy distribution with location parameter x0 and scale parameter gamma.

var v = median( 10.0, 5.0 );
// returns 10.0

v = median( 7.0, 2.0 );
// returns 7.0

If provided NaN as any argument, the function returns NaN.

var v = median( NaN, 5.0 );
// returns NaN

v = median( 20.0, NaN );
// returns NaN

If provided gamma <= 0, the function returns NaN.

var v = median( 1.0, -1.0 );
// returns NaN

v = median( 1.0, 0.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var EPS = require( '@stdlib/constants/float64/eps' );
var median = require( '@stdlib/stats/base/dists/cauchy/median' );

var gamma;
var x0;
var v;
var i;

for ( i = 0; i < 10; i++ ) {
    x0 = randu() * 100.0;
    gamma = ( randu()*10.0 ) + EPS;
    v = median( x0, gamma );
    console.log( 'x0: %d, γ: %d, Median(X;x0,γ): %d', x0.toFixed( 4 ), gamma.toFixed( 4 ), v.toFixed( 4 ) );
}
