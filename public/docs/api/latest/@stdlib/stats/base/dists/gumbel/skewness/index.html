where ζ is the Riemann zeta function.

Usage

var skewness = require( '@stdlib/stats/base/dists/gumbel/skewness' );

skewness( mu, beta )

Returns the skewness for a Gumbel distribution with location parameter mu and scale parameter beta.

var y = skewness( 2.0, 1.0 );
// returns ~1.14

y = skewness( 0.0, 1.0 );
// returns ~1.14

y = skewness( -1.0, 4.0 );
// returns ~1.14

If provided NaN as any argument, the function returns NaN.

var y = skewness( NaN, 1.0 );
// returns NaN

y = skewness( 0.0, NaN );
// returns NaN

If provided beta <= 0, the function returns NaN.

var y = skewness( 0.0, 0.0 );
// returns NaN

y = skewness( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var skewness = require( '@stdlib/stats/base/dists/gumbel/skewness' );

var beta;
var mu;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    beta = randu() * 20.0;
    y = skewness( mu, beta );
    console.log( 'µ: %d, β: %d, skew(X;µ,β): %d', mu.toFixed( 4 ), beta.toFixed( 4 ), y.toFixed( 4 ) );
}
