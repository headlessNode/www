Median

Raised cosine distribution median.

The median for a raised cosine random variable with location parameter mu and scale parameter s is

Usage

var median = require( '@stdlib/stats/base/dists/cosine/median' );

median( mu, s )

Returns the median for a raised cosine distribution with location parameter mu and scale parameter s.

var y = median( 2.0, 1.0 );
// returns 2.0

y = median( 0.0, 1.0 );
// returns 0.0

y = median( -1.0, 4.0 );
// returns -1.0

If provided NaN as any argument, the function returns NaN.

var y = median( NaN, 1.0 );
// returns NaN

y = median( 0.0, NaN );
// returns NaN

If provided s <= 0, the function returns NaN.

var y = median( 0.0, 0.0 );
// returns NaN

y = median( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var median = require( '@stdlib/stats/base/dists/cosine/median' );

var mu;
var s;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    s = randu() * 20.0;
    y = median( mu, s );
    console.log( 'µ: %d, s: %d, Median(X;µ,s): %d', mu.toFixed( 4 ), s.toFixed( 4 ), y.toFixed( 4 ) );
}
