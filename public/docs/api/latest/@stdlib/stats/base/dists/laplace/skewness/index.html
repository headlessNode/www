Skewness

Laplace distribution skewness.

The skewness for a Laplace random variable with location parameter mu and scale parameter b > 0 is

Usage

var skewness = require( '@stdlib/stats/base/dists/laplace/skewness' );

skewness( mu, b )

Returns the skewness for a Laplace distribution with location parameter mu and scale parameter b.

var y = skewness( 2.0, 1.0 );
// returns 0.0

y = skewness( 0.0, 1.0 );
// returns 0.0

y = skewness( -1.0, 4.0 );
// returns 0.0

If provided NaN as any argument, the function returns NaN.

var y = skewness( NaN, 1.0 );
// returns NaN

y = skewness( 0.0, NaN );
// returns NaN

If provided b <= 0, the function returns NaN.

var y = skewness( 0.0, 0.0 );
// returns NaN

y = skewness( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var skewness = require( '@stdlib/stats/base/dists/laplace/skewness' );

var mu;
var b;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    b = randu() * 20.0;
    y = skewness( mu, b );
    console.log( 'µ: %d, b: %d, skew(X;µ,b): %d', mu.toFixed( 4 ), b.toFixed( 4 ), y.toFixed( 4 ) );
}
