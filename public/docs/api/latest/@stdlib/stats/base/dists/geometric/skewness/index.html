Skewness

Geometric distribution skewness.

where p is the success probability.

Usage

var skewness = require( '@stdlib/stats/base/dists/geometric/skewness' );

skewness( p )

Returns the skewness of a geometric distribution with success probability p.

var v = skewness( 0.1 );
// returns ~2.003

v = skewness( 0.5 );
// returns ~2.121

If provided a success probability p outside of [0,1], the function returns NaN.

var v = skewness( NaN );
// returns NaN

v = skewness( 1.5 );
// returns NaN

v = skewness( -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var round = require( '@stdlib/math/base/special/round' );
var skewness = require( '@stdlib/stats/base/dists/geometric/skewness' );

var v;
var i;
var p;

for ( i = 0; i < 10; i++ ) {
    p = randu();
    v = skewness( p );
    console.log( 'p: %d, skew(X;p): %d', p.toFixed( 4 ), v.toFixed( 4 ) );
}
