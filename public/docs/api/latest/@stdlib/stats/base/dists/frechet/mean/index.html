Usage

var mean = require( '@stdlib/stats/base/dists/frechet/mean' );

mean( alpha, s, m )

Returns the expected value for a Fréchet distribution with shape alpha > 0, scale s > 0, and location parameter m.

var y = mean( 2.0, 1.0, 1.0 );
// returns ~2.772

y = mean( 4.0, 4.0, -1.0 );
// returns ~3.902

y = mean( 1.0, 1.0, 2.0 );
// returns Infinity

If provided NaN as any argument, the function returns NaN.

var y = mean( NaN, 1.0, -2.0 );
// returns NaN

y = mean( 1.0, NaN, -2.0 );
// returns NaN

y = mean( 1.0, 1.0, NaN );
// returns NaN

If provided alpha <= 0, the function returns NaN.

var y = mean( 0.0, 3.0, 2.0 );
// returns NaN

y = mean( 0.0, -1.0, 2.0 );
// returns NaN

If provided s <= 0, the function returns NaN.

var y = mean( 1.0, 0.0, 2.0 );
// returns NaN

y = mean( 1.0, -1.0, 2.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var EPS = require( '@stdlib/constants/float64/eps' );
var mean = require( '@stdlib/stats/base/dists/frechet/mean' );

var alpha;
var m;
var s;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    alpha = ( randu()*20.0 ) + EPS;
    s = ( randu()*20.0 ) + EPS;
    m = ( randu()*20.0 ) - 40.0;
    y = mean( alpha, s, m );
    console.log( 'α: %d, s: %d, m: %d, E(X;α,s,m): %d', alpha.toFixed( 4 ), s.toFixed( 4 ), m.toFixed( 4 ), y.toFixed( 4 ) );
}
