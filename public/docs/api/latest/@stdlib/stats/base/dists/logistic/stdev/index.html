Standard Deviation

Logistic distribution standard deviation.

The standard deviation for a logistic random variable with location μ and scale s > 0 is

Usage

var stdev = require( '@stdlib/stats/base/dists/logistic/stdev' );

stdev( mu, s )

Returns the standard deviation for a logistic distribution with location parameter mu and scale parameter s.

var y = stdev( 2.0, 1.0 );
// returns ~1.814

y = stdev( 0.0, 1.0 );
// returns ~1.814

y = stdev( -1.0, 4.0 );
// returns ~7.255

If provided NaN as any argument, the function returns NaN.

var y = stdev( NaN, 1.0 );
// returns NaN

y = stdev( 0.0, NaN );
// returns NaN

If provided s <= 0, the function returns NaN.

var y = stdev( 0.0, 0.0 );
// returns NaN

y = stdev( 0.0, -1.0 );
// returns NaN

Examples

var randu = require( '@stdlib/random/base/randu' );
var stdev = require( '@stdlib/stats/base/dists/logistic/stdev' );

var mu;
var s;
var y;
var i;

for ( i = 0; i < 10; i++ ) {
    mu = ( randu()*10.0 ) - 5.0;
    s = randu() * 20.0;
    y = stdev( mu, s );
    console.log( 'µ: %d, s: %d, SD(X;µ,s): %d', mu.toFixed( 4 ), s.toFixed( 4 ), y.toFixed( 4 ) );
}
