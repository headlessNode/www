lowercase

Convert a string to lowercase.

Usage

var lowercase = require( '@stdlib/string/lowercase' );

lowercase( str )

Converts a string to lowercase.

var str = lowercase( 'bEEp' );
// returns 'beep'

Examples

var lowercase = require( '@stdlib/string/lowercase' );

var str;

str = lowercase( 'Beep' );
// returns 'beep'

str = lowercase( 'BeEp' );
// returns 'beep'

str = lowercase( 'Beep BOOP' );
// returns 'beep boop'

str = lowercase( '$**_Beep_BoOp_**$' );
// returns '$**_beep_boop_**$'

str = lowercase( '' );
// returns ''

CLI

Usage
