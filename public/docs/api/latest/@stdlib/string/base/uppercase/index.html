uppercase

Convert a string to uppercase.

Usage

var uppercase = require( '@stdlib/string/base/uppercase' );

uppercase( str )

Converts a string to uppercase.

var str = uppercase( 'bEEp' );
// returns 'BEEP'

Examples

var uppercase = require( '@stdlib/string/base/uppercase' );

var str = uppercase( 'Beep' );
// returns 'BEEP'

str = uppercase( 'BeEp' );
// returns 'BEEP'

str = uppercase( 'Beep BOOP' );
// returns 'BEEP BOOP'

str = uppercase( '$**_beep_BoOp_**$' );
// returns '$**_BEEP_BOOP_**$'

str = uppercase( '' );
// returns ''
