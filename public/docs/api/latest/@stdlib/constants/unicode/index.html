Unicode Constants

Unicode constants.

Usage

var constants = require( '@stdlib/constants/unicode' );

constants

Namespace containing Unicode constants.

var ns = constants;
// returns {...}
  • MAX_BMP: maximum Unicode code point in the Basic Multilingual Plane (BMP).
  • MAX: maximum Unicode code point.

Examples

var objectKeys = require( '@stdlib/utils/keys' );
var constants = require( '@stdlib/constants/unicode' );

console.log( objectKeys( constants ) );
