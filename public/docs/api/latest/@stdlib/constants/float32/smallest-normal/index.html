Smallest Normal

Smallest positive normalized single-precision floating-point number.

Usage

var FLOAT32_SMALLEST_NORMAL = require( '@stdlib/constants/float32/smallest-normal' );

FLOAT32_SMALLEST_NORMAL

// returns true
