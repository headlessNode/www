Number of Bytes

Size (in bytes) of a 16-bit unsigned integer.

var UINT16_NUM_BYTES = require( '@stdlib/constants/uint16/num-bytes' );

UINT16_NUM_BYTES

var UINT16_NUM_BYTES = require( '@stdlib/constants/uint16/num-bytes' );
