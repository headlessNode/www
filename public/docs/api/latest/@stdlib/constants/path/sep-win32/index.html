Path Separator

Windows path segment separator.

Usage

var PATH_SEP_WIN32 = require( '@stdlib/constants/path/sep-win32' );

PATH_SEP_WIN32

Windows path segment separator.

var sep = PATH_SEP_WIN32;
// returns '\\'

Examples

var PATH_SEP_WIN32 = require( '@stdlib/constants/path/sep-win32' );

var parts;
var path;

path = 'foo\\bar\\baz';
parts = path.split( PATH_SEP_WIN32 );
// returns ['foo','bar','baz']

path = 'foo/bar/baz';
parts = path.split( PATH_SEP_WIN32 );
// returns ['foo/bar/baz' ]
