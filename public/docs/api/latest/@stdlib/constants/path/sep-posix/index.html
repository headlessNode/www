Path Separator

POSIX path segment separator.

Usage

var PATH_SEP_POSIX = require( '@stdlib/constants/path/sep-posix' );

PATH_SEP_POSIX

POSIX path segment separator.

var sep = PATH_SEP_POSIX;
// returns '/'

Examples

var PATH_SEP_POSIX = require( '@stdlib/constants/path/sep-posix' );

var parts;
var path;

path = 'foo/bar/baz';
parts = path.split( PATH_SEP_POSIX );
// returns ['foo','bar','baz']

path = 'foo\\bar\\baz';
parts = path.split( PATH_SEP_POSIX );
// returns ['foo\\bar\\baz' ]
