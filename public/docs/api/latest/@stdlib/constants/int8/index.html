Constants

8-bit signed integer mathematical constants.

Usage

var constants = require( '@stdlib/constants/int8' );

constants

8-bit signed integer mathematical constants.

var c = constants;
// returns {...}
  • MAX: maximum signed 8-bit integer.
  • MIN: minimum signed 8-bit integer.
  • NUM_BYTES: size (in bytes) of an 8-bit signed integer.

Examples

var objectKeys = require( '@stdlib/utils/keys' );
var constants = require( '@stdlib/constants/int8' );

console.log( objectKeys( constants ) );
