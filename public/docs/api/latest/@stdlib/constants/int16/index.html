Constants

16-bit signed integer mathematical constants.

Usage

var constants = require( '@stdlib/constants/int16' );

constants

16-bit signed integer mathematical constants.

var c = constants;
// returns {...}
  • MAX: maximum signed 16-bit integer.
  • MIN: minimum signed 16-bit integer.
  • NUM_BYTES: size (in bytes) of a 16-bit signed integer.

Examples

var objectKeys = require( '@stdlib/utils/keys' );
var constants = require( '@stdlib/constants/int16' );

console.log( objectKeys( constants ) );
