Lucas Number

Maximum safe Lucas number when stored in double-precision floating-point format.

Usage

var FLOAT64_MAX_SAFE_LUCAS = require( '@stdlib/constants/float64/max-safe-lucas' );

FLOAT64_MAX_SAFE_LUCAS

The maximum safe Lucas number when stored in double-precision floating-point format.

var bool = ( FLOAT64_MAX_SAFE_LUCAS === 7639424778862807 );
// returns true

Examples

var FLOAT64_MAX_SAFE_LUCAS = require( '@stdlib/constants/float64/max-safe-lucas' );

var v;
var i;

function lucas( n ) {
    var a;
    var b;
    var c;
    var i;

    a = 2;
    if ( n === 0 ) {
        return a;
    }
    b = 1;
    for ( i = 2; i <= n; i++ ) {
        c = a + b;
        a = b;
        b = c;
    }
    return b;
}

for ( i = 0; i < 100; i++ ) {
    v = lucas( i );
    if ( v > FLOAT64_MAX_SAFE_LUCAS ) {
        console.log( 'Unsafe: %d', v );
    } else {
        console.log( 'Safe:   %d', v );
    }
}
