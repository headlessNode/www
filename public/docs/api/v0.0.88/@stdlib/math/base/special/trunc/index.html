Truncate

Round a numeric value toward zero.

Usage

var trunc = require( '@stdlib/math/base/special/trunc' );

trunc( x )

Rounds a numeric value toward zero.

var v = trunc( -4.2 );
// returns -4.0

v = trunc( 9.99999 );
// returns 9.0

v = trunc( 0.0 );
// returns 0.0

v = trunc( -0.0 );
// returns -0.0

v = trunc( NaN );
// returns NaN

v = trunc( Infinity );
// returns Infinity

v = trunc( -Infinity );
// returns -Infinity

Examples

var randu = require( '@stdlib/random/base/randu' );
var trunc = require( '@stdlib/math/base/special/trunc' );

var x;
var i;

for ( i = 0; i < 100; i++ ) {
    x = (randu()*100.0) - 50.0;
    console.log( 'trunc(%d) = %d', x, trunc( x ) );
}
