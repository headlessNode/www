isArrayArray

Test if a value is an array of arrays.

Usage

var isArrayArray = require( '@stdlib/assert/is-array-array' );

isArrayArray( value )

Tests if a value is an array of arrays.

var bool = isArrayArray( [ [], [] ] );
// returns true

bool = isArrayArray( [ {}, {} ] );
// returns false

bool = isArrayArray( [] );
// returns false

bool = isArrayArray( {} );
// returns false

bool = isArrayArray( [ null, {} ] );
// returns false

Examples

var isArrayArray = require( '@stdlib/assert/is-array-array' );

var bool = isArrayArray( [ [], [], [] ] );
// returns true

bool = isArrayArray( [ [], {} ] );
// returns false

bool = isArrayArray( [] );
// returns false

bool = isArrayArray( {} );
// returns false

bool = isArrayArray( [ 'a', 'b' ] );
// returns false

bool = isArrayArray( null );
// returns false
