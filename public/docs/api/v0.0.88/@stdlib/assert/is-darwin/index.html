IS_DARWIN

Boolean indicating if the current process is running on Darwin.

Usage

var IS_DARWIN = require( '@stdlib/assert/is-darwin' );

IS_DARWIN

Boolean indicating if the current process is running on Darwin.

console.log( IS_DARWIN );
// => <boolean>

Examples

var IS_DARWIN = require( '@stdlib/assert/is-darwin' );

if ( IS_DARWIN ) {
    console.log( 'Running on Darwin...' );
} else {
    console.log( 'Running on %s...', process.platform );
}
